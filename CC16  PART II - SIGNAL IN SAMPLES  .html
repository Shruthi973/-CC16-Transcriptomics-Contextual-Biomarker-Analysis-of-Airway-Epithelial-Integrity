# Pull sample metadata from the Series Matrix header

def read_series_header(path, n_lines=4000):
    header_lines = []
    with open(path, "r", errors="ignore") as f:
        for _ in range(n_lines):
            line = f.readline()
            if not line:
                break
            if line.startswith("!"):
                header_lines.append(line.strip())
            # stopping once we hit the matrix header
            if line.startswith("ID_REF"):
                break
    return header_lines

header_lines = read_series_header(file_path)

# Extract sample characteristics lines (these carry COPD/smoking/FEV1 etc.)
char_lines = [ln for ln in header_lines if ln.startswith("!Sample_characteristics_ch1")]
title_line = [ln for ln in header_lines if ln.startswith("!Sample_title")]

    # returns list of items inside "..."

import re

def grab(patterns, text):
    for pat in patterns:
        m = re.search(pat, text, flags=re.IGNORECASE)
        if m:
            return m.group(1).strip()
    return np.nan

meta["copd_status"] = meta["characteristics_raw"].apply(

        r"disease\s*state:\s*([^|]+)",

    ], s)
)

)

import re
import numpy as np
import pandas as pd

def extract_value(text, key):
    """
    Pulls value from patterns like 'key: value' inside a ' | ' separated string.
    Returns NaN if not found.
    """
    text = str(text)
    # match 'key: something' until next separator '|'
    m = re.search(rf"{re.escape(key)}\s*:\s*([^|]+)", text, flags=re.IGNORECASE)
    return m.group(1).strip() if m else np.nan

# Extract fields that are clearly present in your raw strings
meta["copd_raw"]         = meta["characteristics_raw"].apply(lambda s: extract_value(s, "copd"))
meta["smoking_status"]   = meta["characteristics_raw"].apply(lambda s: extract_value(s, "smoking status"))
meta["age_years"]        = meta["characteristics_raw"].apply(lambda s: extract_value(s, "age (years)"))
meta["sex"]              = meta["characteristics_raw"].apply(lambda s: extract_value(s, "Sex"))
meta["pack_years"]       = meta["characteristics_raw"].apply(lambda s: extract_value(s, "pack years"))
meta["fev1_percent"]     = meta["characteristics_raw"].apply(lambda s: extract_value(s, "fev1%"))
meta["fev1_fvc"]         = meta["characteristics_raw"].apply(lambda s: extract_value(s, "fev1/fvc"))
meta["asthma_history"]   = meta["characteristics_raw"].apply(lambda s: extract_value(s, "history of asthma"))
meta["inhaled_meds"]     = meta["characteristics_raw"].apply(lambda s: extract_value(s, "inhaled medications"))
meta["delta_fev1"]       = meta["characteristics_raw"].apply(lambda s: extract_value(s, "change in fev1"))
meta["used_in_analysis"] = meta["characteristics_raw"].apply(lambda s: extract_value(s, "used in analysis"))

# Clean types
meta["copd_group"] = meta["copd_raw"].astype(str).str.lower().map({"yes": "COPD", "no": "Control"})
